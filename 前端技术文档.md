### 路由守卫

```typescript
router.beforeEach((to, from, next) => {
  const userStore = useUserStore()

  // 需要登录的页面
  if (to.meta.requiresAuth && !userStore.isLoggedIn) {
    next({ name: 'Login', query: { redirect: to.fullPath } })
    return
  }

  // 已登录访问登录页，跳转首页
  if (to.name === 'Login' && userStore.isLoggedIn) {
    next({ name: 'Home' })
    return
  }

  next()
})
```

## 四、状态管理（Pinia）

### 用户状态 store/user.ts

```typescript
interface UserState {
  token: string | null
  userInfo: UserInfo | null
  isLoggedIn: boolean
}

export const useUserStore = defineStore('user', {
  state: (): UserState => ({
    token: null,
    userInfo: null,
    isLoggedIn: false
  }),

  getters: {
    userId: (state) => state.userInfo?.id,
    avatar: (state) => state.userInfo?.avatar || DEFAULT_AVATAR,
    nickname: (state) => state.userInfo?.nickname || '未登录'
  },

  actions: {
    setToken(token: string) {
      this.token = token
      this.isLoggedIn = !!token
    },

    setUserInfo(userInfo: UserInfo) {
      this.userInfo = userInfo
    },

    async login(phone: string, code: string) {
      const res = await authApi.login({ phone, code })
      this.setToken(res.token)
      this.setUserInfo(res.user)
      return res
    },

    async logout() {
      await authApi.logout()
      this.token = null
      this.userInfo = null
      this.isLoggedIn = false
    },

    async refreshUserInfo() {
      if (!this.token) return
      const res = await authApi.getUserInfo()
      this.setUserInfo(res)
    }
  },

  persist: {
    key: 'fishing-user',
    storage: localStorage,
    paths: ['token', 'userInfo', 'isLoggedIn']
  }
})
```

### 草稿状态 store/draft.ts

```typescript
interface DraftState {
  currentDraft: Draft | null
  drafts: Draft[]
}

export const useDraftStore = defineStore('draft', {
  state: (): DraftState => ({
    currentDraft: null,
    drafts: []
  }),

  actions: {
    saveDraft(draft: Partial<Draft>) {
      const index = this.drafts.findIndex(d => d.id === draft.id)
      if (index >= 0) {
        this.drafts[index] = { ...this.drafts[index], ...draft }
      } else {
        this.drafts.push({
          id: Date.now().toString(),
          createTime: new Date(),
          ...draft
        } as Draft)
      }
      this.currentDraft = this.drafts[this.drafts.length - 1]
      this.saveToLocal()
    },

    deleteDraft(id: string) {
      this.drafts = this.drafts.filter(d => d.id !== id)
      if (this.currentDraft?.id === id) {
        this.currentDraft = null
      }
      this.saveToLocal()
    },

    clearDraft() {
      this.currentDraft = null
    },

    saveToLocal() {
      localStorage.setItem('fishing-drafts', JSON.stringify(this.drafts))
    },

    loadFromLocal() {
      const data = localStorage.getItem('fishing-drafts')
      if (data) {
        this.drafts = JSON.parse(data)
      }
    }
  }
})
```

### 请求封装 api/index.ts

```typescript
import axios from 'axios'
import { showToast } from 'vant'
import type { AxiosRequestConfig, AxiosResponse } from 'axios'

const baseURL = import.meta.env.VITE_API_BASE_URL

const request = axios.create({
  baseURL,
  timeout: 15000,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
request.interceptors.request.use(
  (config) => {
    const userStore = useUserStore()
    if (userStore.token) {
      config.headers.Authorization = `Bearer ${userStore.token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// 响应拦截器
request.interceptors.response.use(
  (response: AxiosResponse) => {
    const { code, message, data } = response.data

    if (code === 0) {
      return data
    }

    // Token失效
    if (code === 401) {
      const userStore = useUserStore()
      userStore.logout()
      showToast('登录已过期，请重新登录')
      router.push({ name: 'Login' })
      return Promise.reject(new Error(message))
    }

    showToast(message || '请求失败')
    return Promise.reject(new Error(message))
  },
  (error) => {
    if (error.response) {
      const status = error.response.status
      const message = {
        400: '请求参数错误',
        401: '未授权，请登录',
        403: '拒绝访问',
        404: '请求资源不存在',
        500: '服务器错误',
        502: '网关错误',
        503: '服务不可用',
        504: '网关超时'
      }[status] || '请求失败'

      showToast(message)
    } else if (error.message.includes('timeout')) {
      showToast('请求超时')
    } else {
      showToast('网络错误')
    }

    return Promise.reject(error)
  }
)

export default request

// 导出带类型的请求方法
export const http = {
  get<T = any>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return request.get(url, config)
  },
  post<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return request.post(url, data, config)
  },
  put<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return request.put(url, data, config)
  },
  delete<T = any>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return request.delete(url, config)
  }
}
```

### 请求钩子 composables/useRequest.ts

```typescript
import { ref } from 'vue'
import { showToast, showLoadingToast, closeToast } from 'vant'

export function useRequest<T = any, P = any>(
  fn: (params: P) => Promise<T>,
  options: {
    loading?: boolean
    successMessage?: string | ((data: T) => string)
    errorMessage?: string
    onSuccess?: (data: T) => void
    onError?: (error: Error) => void
  } = {}
) {
  const loading = ref(false)
  const data = ref<T>() as Ref<T>
  const error = ref<Error>()

  const execute = async (params: P) => {
    try {
      loading.value = true
      error.value = undefined

      if (options.loading) {
        showLoadingToast({ message: '加载中...', forbidClick: true })
      }

      const res = await fn(params)
      data.value = res

      if (options.successMessage) {
        const message = typeof options.successMessage === 'function'
          ? options.successMessage(res)
          : options.successMessage
        showToast({ type: 'success', message })
      }

      options.onSuccess?.(res)
      return res
    } catch (e) {
      error.value = e as Error
      if (options.errorMessage) {
        showToast({ type: 'fail', message: options.errorMessage })
      }
      options.onError?.(e as Error)
      throw e
    } finally {
      loading.value = false
      closeToast()
    }
  }

  return {
    loading,
    data,
    error,
    execute
  }
}
```

```typescript
import { ref } from 'vue'
```

```typescript
export function useLocation() {
  const loading = ref(false)
  const location = ref<{ lng: number; lat: number; address: string }>()
  const error = ref<string>()

  // 获取当前位置
  const getCurrentLocation = async () => {
    loading.value = true
    error.value = undefined

    try {
      // 使用浏览器定位
      const position = await new Promise<GeolocationPosition>((resolve, reject) => {
        navigator.geolocation.getCurrentPosition(resolve, reject, {
          enableHighAccuracy: true,
          timeout: 10000
        })
      })

      const { longitude, latitude } = position.coords

      // 逆地理编码
      const AMap = await AMapLoader.load({
        key: import.meta.env.VITE_AMAP_KEY,
        version: '2.0',
        plugins: ['Geocoder']
      })

      const geocoder = new AMap.Geocoder()
      const result = await new Promise((resolve, reject) => {
        geocoder.getAddress([longitude, latitude], (status: string, result: any) => {
          if (status === 'complete') {
            resolve(result)
          } else {
            reject(new Error('逆地理编码失败'))
          }
        })
      })

      location.value = {
        lng: longitude,
        lat: latitude,
        address: result.regeocode.formattedAddress
      }

      return location.value
    } catch (e: any) {
      error.value = e.message
      throw e
    } finally {
      loading.value = false
    }
  }

  // 计算距离
  const calculateDistance = (
    lng1: number,
    lat1: number,
    lng2: number,
    lat2: number
  ): number => {
    const R = 6371 // 地球半径
    const dLat = (lat2 - lat1) * Math.PI / 180
    const dLng = (lng2 - lng1) * Math.PI / 180
    const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
      Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
      Math.sin(dLng / 2) * Math.sin(dLng / 2)
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
    return R * c
  }

  // 格式化距离
  const formatDistance = (meters: number): string => {
    if (meters < 1000) {
      return `${Math.round(meters)}m`
    }
    return `${(meters / 1000).toFixed(1)}km`
  }

  return {
    loading,
    location,
    error,
    getCurrentLocation,
    calculateDistance,
    formatDistance
  }
}
```

```vue
    video: contentType.value === 'video' ? video.value : undefined,
    locationId: selectedSpot.value?.id
  }

  await submitFeed(data)
  draftStore.clearDraft()
  router.back()
}

const onCancel = async () => {
  if (content.value.length > 0) {
    const confirmed = await showConfirmDialog({
      title: '确认退出',
      message: '内容未发布，是否保存为草稿？'
    })

    if (confirmed) {
      draftStore.saveDraft({
        contentType: contentType.value,
        content: content.value,
        category: category.value,
        images: images.value,
        video: video.value,
        selectedSpot: selectedSpot.value
      })
    }
  }

  router.back()
}

// 检查是否有草稿
onMounted(() => {
  if (draftStore.currentDraft) {
    showConfirmDialog({
      title: '发现草稿',
      message: '您有未完成的草稿，是否继续编辑？'
    }).then((confirm) => {
      if (confirm) {
        const draft = draftStore.currentDraft!
        contentType.value = draft.contentType
        content.value = draft.content
        category.value = draft.category
        images.value = draft.images || []
        video.value = draft.video || null
        selectedSpot.value = draft.selectedSpot || null
      } else {
        draftStore.clearDraft()
      }
    })
  }
})
</script>
```
